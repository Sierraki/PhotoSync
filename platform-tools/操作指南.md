# CMD
### 1. 准备工作
cd /d C:\Users\Sierra\Desktop\platform-tools
adb devices

### 2. 推送文件 (电脑 -> 手机)
adb -s R5CX719S69R push "S:\手机备份\新建文件夹\Basic" "/sdcard/DCIM/"

### 3. 拉取文件 (手机 -> 电脑)
adb -s R5CX719S69R pull "/sdcard/DCIM/ELSE" "C:\Users\Sierra\Desktop\done"
adb -s R5CX719S69R pull "/sdcard/DCIM/Basic" "C:\Users\Sierra\Desktop\done"
adb -s R5CX719S69R pull "/sdcard/DCIM/Wallpaper" "C:\Users\Sierra\Desktop\done"
adb -s R5CX719S69R pull "/sdcard/DCIM/Screenshots" "C:\Users\Sierra\Desktop\done\新建文件夹"

### 4. 强制刷新相册与重启
adb -s R5CX719S69R shell "mv /sdcard/DCIM/Camera /sdcard/DCIM/Camera_tmp && mv /sdcard/DCIM/Camera_tmp /sdcard/DCIM/Camera"
adb -s R5CX719S69R reboot
